CREATE SERVER AUDIT [SuccessfulLogins]
TO FILE 
(	FILEPATH = N'E:\SQL_LOGS'
	,MAXSIZE = 500 MB
	,MAX_FILES = 10
	,RESERVE_DISK_SPACE = OFF
)
WITH
(	QUEUE_DELAY = 1000
	,ON_FAILURE = CONTINUE
)

GO

GO

CREATE SERVER AUDIT SPECIFICATION [ServerAuditSpecification-SuccessfulLogins]
FOR SERVER AUDIT [SuccessfulLogins]
ADD (SUCCESSFUL_LOGIN_GROUP)

GO
